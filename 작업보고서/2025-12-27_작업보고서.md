## 6. 참고 명령어

```bash
# today 저장소 작업
cd C:\todo\today
git add .
git commit -m "커밋 메시지"
git push

# AWS EC2 접속 (UTTEC Edu)
ssh -i "uttec-first-ec2.pem" ec2-user@13.125.148.58

# PM2 관리
pm2 list
pm2 restart hw-c-edu-frontend
pm2 logs hw-c-edu-frontend

# 빌드 및 배포
cd /home/ec2-user/hw-c-edu-platform/frontend
npm run build
pm2 restart hw-c-edu-frontend

# Nginx 관련
sudo nginx -t                    # 설정 테스트
sudo systemctl reload nginx      # 설정 재적용
sudo systemctl status nginx      # 상태 확인

# SSL 인증서 재발급 (DuckDNS 문제 해결 후)
sudo certbot --nginx -d uttec-edu.duckdns.org --non-interactive --agree-tos --email admin@uttec.com --redirect
```
